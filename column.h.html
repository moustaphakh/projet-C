    int *values;
    size_t size;
}COLUMN;

int countValuesGreaterThanX(int array[], int size, int x);
int countValuesLessThanX(int array[], int size, int x);
int countValuesEqualToX(int array[], int size, int x);




#endif //PROJET_C_COLUMN_H
